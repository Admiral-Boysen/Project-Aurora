	id = ausgleich.2
	title = ausgleich.2.t
	desc = ausgleich.2.d
	
	is_triggered_only = yes
	
	option = {
		name = ausgleich.2.a #Stay Dualism

		ai_chance = { factor = 40 }

		every_country = {
			limit = {
				has_idea = austria_dualism
			}

			add_opinion_modifier = { modifier = dualism_disagree }
		}

		GER = { add_opinion_modifier = {target = AUS modifier = dualism_agree} }
		add_opinion_modifier = {target = GER modifier = dualism_agree}

		HUN = {country_event = { id = ausgleich.3 days = 1 }}	
	}

	option = {
		name = ausgleich.2.b #Federalism

		ai_chance = { factor = 30 }

		every_country = {
			limit = {
				has_idea = austria_dualism
			}

			add_opinion_modifier = { modifier = federalism_agree }
		}

		GER = { add_opinion_modifier = {target = AUS modifier = federlism_disagree}}

		HUN = {country_event = { id = ausgleich.4 days = 1 }}		
	}
	
	option = {
		name = ausgleich.2.c #Centralize

		ai_chance = { factor = 25 }

		every_country = {
			limit = {
				has_idea = austria_dualism
			}
