### Ausgleich Event for Austria
### Decides what course to take
Country_event = {
	id = ausgleich.2
	title = ausgleich.2.t
	desc = ausgleich.2.d
	
	is_triggered_only = yes
	
	option = {
		name = ausgleich.2.a #Stay Dualism
		every_country = {
			limit = {
				has_idea = austria_dualism
			}
			add_opinion_modifier = {target = this modifier = dualism_disagree}
		}
		GER = { add_opinion_modifier = {target = AUS modifier = dualism_agree} }
		add_opinion_modifier = {target = GER modifier = dualism_agree}
		HUN = {country_event = { id = ausgleich.3 days = 1 }}
		
		ai_chance = {
			base = 30
		}
	}

	option = {
		name = ausgleich.2.b #Federalism
		every_country = {
			limit = {
				has_idea = austria_dualism
			}
			add_opinion_modifier = {target = this modifier = federalism_agree}
		}
		GER = { add_opinion_modifier = {target = AUS modifier = federlism_disagree}}
		HUN = {country_event = { id = ausgleich.4 days = 1 }}
		
		ai_chance = {
			base = 20
		}
	}
	
	option = {
		name = ausgleich.2.c #Centralize
		every_country = {
			limit = {
				has_idea = austria_dualism
			}
			add_opinion_modifier = {target = this modifier = centralize_disagree}
		}
		HUN = {country_event = { id = ausgleich.5 days = 1 }}
		
		ai_chance = {
			base = 50
		}
	}
}

	title = ausgleich.7.t
	desc = ausgleich.7.d
	
	is_triggered_only = yes
	
	option = {
		name = ausgleich.7.a
		BOM = { country_event = {id = ausgleich.8 days = 1}}
		CRO = { country_event = {id = ausgleich.8 days = 1}}
		GAL = { country_event = {id = ausgleich.8 days = 1}}
		BOS = { country_event = {id = ausgleich.8 days = 1}}
		HUN = { country_event = {id = ausgleich.8 days = 1}}
		HUN = { release = CRO}
		AUS = { puppet = CRO}
		
		DNF = { transfer_state = 152 }
		DNF = { change_tag_from = AUS }
		DNF = { annex_country = { target = AUS transfer_troops = yes }}
		DNF = {country_event = {id = ausgleich.9 days = 2}}
		
		every_state = {
			limit = {
				OR = {
					is_core_of = BOM
					is_core_of = CRO
					is_core_of = GAL
					is_core_of = BOS
					is_core_of = HUN
				}
			}
			DNF = {
				add_state_core = PREV
			}
		}
	}
	
}
