# Imperial Rule Assistance Society
country_event = {
	id = kr.japan.1
	title = kr.japan.1.t
	desc = kr.japan.1.d
	#DH Picture "Japan_Victory2"
	
	fire_only_once = yes
	
	trigger = {
		tag = JAP
		has_war = yes
		NOT = {
			OR = {
				has_government = totalist
				has_government = radical_socialist
				has_government = syndicalist
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1 
	}
	
	option = {
		name = kr.japan.1.a #"Glory to the Rising Sun! Banzai! Banzai!"
		add_ideas = japan_parties_banned
		retire_country_leader = yes
		set_politics = {
			ruling_party = authoritarian_democratist
			elections_allowed = no
		}
		add_popularity = {
			ideology = authoritarian_democratist
			popularity = 1.0
		}
		set_party_name = {
			ideology = authoritarian_democratist
			long_name = JAP_authoritarian_democratist_party_long2
			name = JAP_authoritarian_democratist_party2
		}
		retire_country_leader = yes
		create_country_leader = {
			name = "Hirohito"
			desc = "POLITICS_HIROHITO_DESC"
			picture = "Portrait_Japan_Hirohito.dds"
			expire = "1965.1.1"
			ideology = authoritarian_democratist_subtype
			traits = {}
		}
	}
}

### Niigata Riots ###
# Civil Disobedience in Niigta
country_event = {
	id = kr.japan.2
	title = kr.japan.2.t
	desc = kr.japan.2.d
	#DH Picture = "Rodonominto"

	fire_only_once = yes
	
	trigger = {
		tag = JAP
		date > 1936.04.03
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = kr.japan.2.a #"Negotiate with the dissenters"
		country_event = {
			id = kr.japan.3
			days = 10
		}
		add_political_power = -20
	}
	option = {
		name = kr.japan.2.b #"Crush this riot"
		add_manpower = -10000
		add_national_unity = -0.10
		send_equipment = {
			target = JAP
			type = infantry_equipment
			amount = -1000
		}
	}
}
