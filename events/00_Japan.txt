# Imperial Rule Assistance Society
country_event = {
	id = japan.1
	title = japan.1.t
	desc = japan.1.d
	#DH Picture "Japan_Victory2"
	
	fire_only_once = yes
	
	trigger = {
		tag = JAP
		has_war = yes
		NOT = {
			OR = {
				has_government = totalist
				has_government = radical_socialist
				has_government = syndicalist
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1 
	}
	
	option = {
		name = japan.1.a
		add_ideas = japan_parties_banned
		retire_country_leader = yes
		set_politics = {
			ruling_party = authoritarian_democratist
			elections_allowed = no
		}
		add_popularity = {
			ideology = authoritarian_democratist
			popularity = 1.0
		}
		set_party_name = {
			ideology = authoritarian_democratist
			long_name = JAP_authoritarian_democratist_party_long2
			name = JAP_authoritarian_democratist_party2
		}
		retire_country_leader = yes
		create_country_leader = {
			name = "Hirohito"
			desc = "POLITICS_HIROHITO_DESC"
			picture = "Portrait_Japan_Hirohito.dds"
			expire = "1965.1.1"
			ideology = authoritarian_democratist_subtype
			traits = {}
		}
	}
}
