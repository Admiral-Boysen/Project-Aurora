### New Chairman

# The Resignation of Philip Snowden

country_event = {
	id = unionofbritain.6
	title = unionofbritain.6.t
	desc = unionofbritain.6.d
	picture = GFX_report_crowd_in_prague
	
	fire_only_once = yes
	
	trigger = {
		tag = ENG
		AND = {
			OR = { has_idea = ENG_tuc_foreign_glaise_idea has_idea = ENG_tuc_foreign_fem_idea has_idea = ENG_tuc_foreign_mosley_idea }	
			OR = { has_idea = ENG_tuc_military_mosley_idea has_idea = ENG_tuc_military_glaise_idea has_idea = ENG_tuc_military_fem_idea  }
			OR = { has_idea = ENG_tuc_internal_mosley_idea has_idea = ENG_tuc_internal_glaise_idea has_idea = ENG_tuc_internal_fem_idea  }
			OR = { has_idea = ENG_tuc_economy_mosley_idea has_idea = ENG_tuc_economy_glaise_idea has_idea = ENG_tuc_economy_fem_idea  }	
		}
	}
	
	mean_time_to_happen = {
		days = 10
	}
	
	option = {
		name = unionofbritain.6.a
	}
}

# A riddle

country_event = {
	id = unionofbritain.11
	title = unionofbritain.11.t
	desc = unionofbritain.11.d
	picture = GFX_report_event_generic_sign_treaty2
	
	trigger = {
		tag = ENG
		has_country_flag = ENG_brr
	}
	
	mean_time_to_happen = {
		days = 500
	}
	
	option = {
		name = unionofbritain.11.a
		hidden_effect = {
			clr_country_flag = ENG_brr 
			set_country_flag = ENG_find_him
		}
	}
	
	option = {
		name = unionofbritain.11.b	
		hidden_effect = {
			clr_country_flag = ENG_brr
		}
	}
}
