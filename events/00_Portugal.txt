## Indian Nations Seizing Goa ##
country_event = {
	id = portugal.7
	title = portugal.7.t
	desc = portugal.7.d
	
	fire_only_once = yes
	
	trigger = {
		tag = PRF
		POR = { exists = yes }
		425 = { is_controlled_by = PRF }
		321 = { is_controlled_by = POR }
		or = {
			POR = {
				has_capitulated = yes
			}
			POR = {
				has_war = yes 
			}
		}
		not = {
			POR = {
				is_in_faction_with = PRF
			}
			POR = {
				is_puppet_of = PRF
			}
			PRF = {
				has_war = yes
			}
		}
	}
	
	option = {
		name = portugal.7.a
		ai_chance = { factor = 90 }
		PRF = { add_state_claim = 321 }
		add_opinion_modifier = {target = POR modifier = seized_goa }
		POR = { country_event = { id = portugal.8 days = 1}}
	}
	
	
	option = {
		name = portugal.7.b
		ai_chance = { factor = 10 }
		add_opinion_modifier = {target = POR modifier = ignored_goa }
	}
}
